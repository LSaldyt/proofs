\problem{Subset Transitivity}

Prove that if $A \subseteq B$, and $B \subseteq C$, then $A \subseteq C$.

\solution

\part 

\begin{theorem}
If $A \subseteq B$, and $B \subseteq C$, then $A \subseteq C$.
\end{theorem}

\begin{proof}
    If $A \subseteq B$, for every element $x \in A$, $x \in B$.
    Similarly, if $B \subseteq C$, for every element $x \in B$, $x \in C$.
    Therefore, for every element $x \in A$, $x \in C$.
\end{proof}
