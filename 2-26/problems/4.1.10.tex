\problem{4.1.10}

Indicate whether the following relations on the given sets are reflexive, symmetric, anti-symmetric, or transitive. 

\solution

\part 

\begin{enumerate}
    \item $A = \{p : p \text{ is a person in alaska}\}$. $x ~ y$ if $x$ is at least as tall as $y$. \\
        Reflexive, not symmetric, anti-symmetric, and transitive.
    \item $A = \mathcal{N}$. $x ~ y$ if $x + y$ is even. \\
        Reflexive, symmetric, not anti-symmetric, and transitive (though somewhat vacuously).
    \item $A = \mathcal{N}$. $x ~ y$ if $x + y$ is odd. \\
        Reflexive, symmetric, not anti-symmetric, and transitive, as above.
    \item $A = \mathcal{P}(\mathcal{N})$. $x ~ y$ if $x \subseteq y$. \\
        Reflexive, not symmetric, anti-symmetric, and transitive.
    \item $A = \mathcal{R}$. $x ~ y$ if $x = 2y$. \\
        Not reflexive, not symmetric, not anti-symmetric, and not transitive.
    \item $A = \mathcal{R}$. $x ~ y$ if $x - y$ is irrational. \\
        Not reflexive, symmetric, not anti-symmetric, and not transitive.
    \item $A = \{l : l \text{is a line in the Cartesian plane}\}$. $x ~ y$ if $x$ and $y$ are parallel lines or if $x$ and $y$ are the same line. \\
        Reflexive, symmetric, not anti-symmetric, and transitive.
\end{enumerate}
