\problem{5.1.12}

Give an example of a function $f \colon \mathcal{R} \to \mathcal{R}$ in which:
\begin{enumerate}
\item $f$ is one-to-one but not onto.
\item $f$ is onto but not one-to-one.
\item $f$ is both one-to-one and onto.
\item $f$ is neither one-to-one nor onto.
\end{enumerate}

\solution

\part 

\begin{enumerate}
\item $f(x) = \sqrt{x}$
\item $f(x) = x^3$
\item $f(x) = x + \pi$
\item $f(x) = sin(x)$
\end{enumerate}
